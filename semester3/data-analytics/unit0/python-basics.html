# Function that calculates and returns a value
def calculate_grade_average(grades):
    """
    Calculate the average of a list of grades.
    
    Args:
        grades (list): List of numerical grades
        
    Returns:
        float: Average grade rounded to 2 decimal places
    """
    if len(grades) == 0:
        return 0.0
    
    total = sum(grades)
    average = total / len(grades)
    return round(average, 2)

# Using the function
student_grades = [85, 92, 78, 96, 88]

print(f"Average grade: {avg_grade}")

# Function with multiple return values
def analyze_grades(grades):
    """Analyze a list of grades and return multiple statistics."""
    if not grades:
        return 0, 0, 0, 0
    
    average = sum(grades) / len(grades)
    highest = max(grades)
    lowest = min(grades)
    passing_count = len([g for g in grades if g >= 70])
    
    return average, highest, lowest, passing_count

# Unpack multiple return values

# Customer analysis functions
def calculate_customer_lifetime_value(monthly_revenue, months_active, retention_rate):
    """
    Calculate Customer Lifetime Value (CLV).
    
    Args:
        monthly_revenue (float): Average monthly revenue per customer
        months_active (int): Number of months customer has been active
        retention_rate (float): Percentage of customers retained (0-1)
    
    Returns:
        float: Estimated Customer Lifetime Value
    """
    if retention_rate <= 0 or retention_rate > 1:
        return 0
    
    # Simple CLV formula: (Monthly Revenue √ó Months Active) / (1 - Retention Rate)
    clv = (monthly_revenue * months_active) / (1 - retention_rate)
    return round(clv, 2)

def categorize_customer_value(clv):

def clean_phone_number(phone):
    """
    Clean and standardize phone number format.
    
    Args:
        phone (str): Phone number in various formats
        
    Returns:
        str: Standardized phone number (XXX) XXX-XXXX
    """
    # Remove all non-digit characters
    digits_only = ''.join(filter(str.isdigit, phone))
    
    # Check if we have 10 or 11 digits
    if len(digits_only) == 11 and digits_only.startswith('1'):
        digits_only = digits_only[1:]  # Remove leading 1
    
    if len(digits_only) == 10:
        # Format as (XXX) XXX-XXXX
        return f"({digits_only[:3]}) {digits_only[3:6]}-{digits_only[6:]}"
    else:
        return "Invalid phone number"

def validate_email(email):
    """
    Basic email validation.
    
    Args:
        email (str): Email address to validate
        
    Returns:
        bool: True if email appears valid, False otherwise
    """
    if '@' not in email:
        return False
    
    parts = email.split('@')
    if len(parts) != 2:
        return False
    
    username, domain = parts
    if len(username) == 0 or len(domain) == 0:
        return False
    
    if '.' not in domain:
        return False
    
    return True

def clean_customer_data(customer_record):
    """
    Clean a customer data record.
    
    Args:
        customer_record (dict): Raw customer data
        
    Returns:
        dict: Cleaned customer data with validation flags
    """
    cleaned = customer_record.copy()
    
    # Clean phone number
    if 'phone' in cleaned:
        cleaned['phone'] = clean_phone_number(cleaned['phone'])
        cleaned['phone_valid'] = cleaned['phone'] != "Invalid phone number"
    
    # Validate email
    if 'email' in cleaned:
        cleaned['email_valid'] = validate_email(cleaned['email'])
    
    # Clean name (title case)
    if 'name' in cleaned:
        cleaned['name'] = cleaned['name'].strip().title()
    
    # Clean city (title case)
    if 'city' in cleaned:
        cleaned['city'] = cleaned['city'].strip().title()
    
    return cleaned
